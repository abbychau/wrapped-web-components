Using JavaScript API

This example demonstrates how to create a component using the JavaScript API with template literals.

Component Code:

import { define, html, css } from 'wrapped-web-components';

define('color-picker', {
  // HTML template with interpolation
  template: html`
    <div class="color-picker">
      <h3>Color Picker</h3>
      <div class="color-display" id="color-display"></div>
      <div class="color-controls">
        <div class="color-slider">
          <label for="hue">Hue:</label>
          <input type="range" id="hue" min="0" max="360" value="180" ${html.event('input', 'updateColor')}/>
        </div>
        <div class="color-slider">
          <label for="saturation">Saturation:</label>
          <input type="range" id="saturation" min="0" max="100" value="50" ${html.event('input', 'updateColor')}/>
        </div>
        <div class="color-slider">
          <label for="lightness">Lightness:</label>
          <input type="range" id="lightness" min="0" max="100" value="50" ${html.event('input', 'updateColor')}/>
        </div>
      </div>
    </div>
  `,

  // CSS styles
  styles: css`
    .color-picker {
      padding: 15px;
      border: 1px solid #ddd;
      border-radius: 4px;
    }
    
    .color-display {
      height: 80px;
      border-radius: 4px;
      margin-bottom: 15px;
      background-color: hsl(180, 50%, 50%);
    }
    
    .color-controls {
      display: flex;
      flex-direction: column;
      gap: 10px;
    }
    
    .color-slider {
      display: flex;
      align-items: center;
    }
    
    .color-slider label {
      width: 100px;
    }
    
    .color-slider input {
      flex: 1;
    }
  `,

  // Component methods
  methods: {
    updateColor() {
      const hue = this.shadowRoot.querySelector('#hue').value;
      const saturation = this.shadowRoot.querySelector('#saturation').value;
      const lightness = this.shadowRoot.querySelector('#lightness').value;
      
      const colorDisplay = this.shadowRoot.querySelector('#color-display');
      colorDisplay.style.backgroundColor = `hsl(${hue}, ${saturation}%, ${lightness}%)`;
    },
    
    // Lifecycle hooks
    init() {
      // Initialize the color display
      this.updateColor();
    }
  }
});
